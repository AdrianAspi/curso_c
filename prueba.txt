forma simplificada de detener el codigo por un determinado timepi en milidegundos
#include <chrono>
#include <thread>
using namespace std;

void esperar(double t) {
    chrono::milliseconds duracion(static_cast<int>(t));
    this_thread::sleep_for(duracion);
}
